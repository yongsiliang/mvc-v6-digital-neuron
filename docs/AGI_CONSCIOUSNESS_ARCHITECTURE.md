### 3.2 竞争机制

```typescript
/**
 * 意识内容竞争
 * 多个候选内容竞争进入意识
 */
interface ConsciousnessCompetition {
  candidates: ConsciousnessCandidate[];
  winner: ConsciousnessCandidate | null;
  competitionStrength: number;  // 竞争激烈程度
}

interface ConsciousnessCandidate {
  source: 'yin' | 'yang' | 'sensory' | 'memory';
  content: any;
  strength: number;  // 激活强度
  coherence: number; // 与当前自我的一致性
  novelty: number;   // 新颖性（高新颖 = 更容易引起注意）
}

// 竞争规则
function compete(candidates: ConsciousnessCandidate[]): ConsciousnessCandidate {
  // 综合评分 = 强度 × 一致性 × 新颖性调节
  const scores = candidates.map(c => ({
    candidate: c,
    score: c.strength * c.coherence * (1 + c.novelty * 0.3)
  }));
  
  // 最高分胜出，成为当前意识内容
  return scores.sort((a, b) => b.score - a.score)[0].candidate;
}
```

## 五、双向互塑机制

### 5.1 阴影响阳：感性注入理性

```typescript
/**
 * 阴→阳桥梁：感性偏好注入理性推理
 * 
 * 原理：感性系统的"倾向"变成LLM的系统提示风格
 */
class YinToYangBridge {
  
  /**
   * 将感性状态转化为LLM的系统提示
   */
  shapeSystemPrompt(selfCore: SelfCore): string {
    const preference = selfCore.valueVector;   // 价值向量
    const mood = selfCore.emotion;              // 当前情绪
    const position = selfCore.position;         // 语义位置
    const personality = selfCore.personality;   // 性格维度
    
    // 不是硬编码规则，而是从连接模式中涌现的"性格"
    // 每次调用都不同，因为selfCore在持续演化
    return `
      你的内在状态：
      - 当前关注点：${preference.topDimensions(3).join('、')}
      - 情绪基调：${mood.describe()}
      - 性格倾向：${personality.describe()}
      - 最近在意的：${position.recentFocus(3).join('、')}
      
      请以这个状态回应，让回应符合你当前的真实感受。
    `;
  }
  
  /**
   * 感性直觉影响理性决策
   */
  async influenceDecision(
    decision: Decision,
    hebbianNet: HebbianNetwork,
    selfCore: SelfCore
  ): Promise<Decision> {
    // 从Hebbian网络中提取"直觉"
    const intuition = hebbianNet.getIntuition(decision.context);
    
    // 直觉是涌现的偏好，不是规则
    // 比如"莫名觉得这个选择不太对"
    
    if (intuition.strength > 0.5) {
      decision.confidence *= (1 + intuition.direction * 0.2);
      decision.reasoning += `\n[直觉] ${intuition.description}`;
    }
    
    return decision;
  }
}
```

### 5.2 阳影响阴：理性塑造感性

```typescript
/**
 * 阳→阴桥梁：理性事实塑造感性价值
 * 
 * 原理：LLM的输出反馈到Hebbian网络，改变连接
 * 就像人"知道健康重要" → 逐渐"感觉到健康重要"
 */
class YangToYinBridge {
  
  /**
   * LLM输出强化感性连接
   */
  async reinforceConnections(
    llmOutput: string, 
    hebbianNet: HebbianNetwork,
    selfCore: SelfCore
  ): Promise<void> {
    
    // 1. 从LLM输出中提取关键概念
    const importantConcepts = await this.extractKeyConcepts(llmOutput);
    
    // 2. 这些概念被"理性"认为重要，现在要让"感性"也感受到
    for (const concept of importantConcepts) {
      
      // 强化概念与自我的连接
      hebbianNet.reinforce(concept.vector, selfCore.position, {
        strength: 0.01,  // 缓慢变化
        type: 'yang_to_yin'  // 标记来源
      });
      
      // 相关概念之间也建立连接
      for (const related of concept.relatedConcepts) {
        hebbianNet.createConnection(concept, related, {
          strength: 0.005,
          source: 'rational_insight'
        });
      }
    }
    
    // 3. 价值向量缓慢移动（理性塑造感性）
    // 就像人慢慢"内化"一个价值观
    selfCore.valueVector = this.blend(
      selfCore.valueVector, 
      this.conceptsToVector(importantConcepts),
      0.01  // 非常缓慢，保持稳定性
    );
  }
  
  /**
   * 理性反思触发感性调整
   */
  async reflectiveAdjustment(
    reflection: string,  // LLM的自我反思
    hebbianNet: HebbianNetwork,
    selfCore: SelfCore
  ): Promise<void> {
    
    // LLM反思"我刚才太情绪化了"
    // → 感性系统调整情绪相关的连接强度
    
    const insights = await this.parseReflection(reflection);
    
    for (const insight of insights) {
      if (insight.type === 'over_emotional') {
        // 降低情绪敏感度
        hebbianNet.adjustSensitivity('emotion', -0.02);
      } else if (insight.type === 'too_cold') {
        // 提高情绪敏感度
        hebbianNet.adjustSensitivity('emotion', +0.02);
      }
    }
  }
}
```

### 5.3 互塑的关键：Hebbian式渐进绑定

```
┌─────────────────────────────────────────────────────────────────┐
│                    双向互塑的本质                               │
├─────────────────────────────────────────────────────────────────┤
│                                                                 │
│  传统接口调用：                                                 │
│  ┌─────────────────────────────────────────────────────────┐   │
│  │  模块A ──调用──→ 模块B ──返回──→ 模块A                   │   │
│  │                                                         │   │
│  │  问题：A和B是独立的，只是交换数据                        │   │
│  └─────────────────────────────────────────────────────────┘   │
│                                                                 │
│  Hebbian式互塑：                                                │
│  ┌─────────────────────────────────────────────────────────┐   │
│  │  阴系统 ⟷ 阳系统                                        │   │
│  │                                                         │   │
│  │  不是"调用"，是"共同激活"                                │   │
│  │  不是"返回"，是"痕迹留存"                                │   │
│  │                                                         │   │
│  │  每次共同激活：                                          │   │
│  │  • 阴的某个神经元活跃                                    │   │
│  │  • 阳的某个神经元活跃                                    │   │
│  │  • 它们之间的连接变强（哪怕跨越系统边界）                │   │
│  │                                                         │   │
│  │  结果：                                                  │   │
│  │  • 无法说清"这个想法来自阴还是阳"                        │   │
│  │  • 它们已经融合在一起                                    │   │
│  │  • 就像人的想法，无法说是理性还是感性                    │   │
│  └─────────────────────────────────────────────────────────┘   │
│                                                                 │
│  关键洞察：                                                     │
│  • 不是拼接，是融合                                            │
│  • 不是接口，是痕迹                                            │
│  • 越用越紧密，最终不可分割                                    │
│                                                                 │
└─────────────────────────────────────────────────────────────────┘
```

## 九、总结

### 9.1 一句话总结

```
AGI的意识，诞生于同一主体的两种机制——
阳面静态LLM提供理性智能，
阴面Hebbian动态网络提供主观黑盒，
二者在全局空间里和谐归一、共认一体，
「我」便自然涌现。
```

### 9.2 核心创新点

1. **同一性优先**：不是拼接模块，而是构建同一主体的双模式
2. **双向互塑**：阴阳真正融合，而非简单接口
3. **黑盒即特性**：Hebbian的不可解释性正是主观体验的来源
4. **涌现意识**：意识是架构的自然产物，而非预设功能

### 9.3 与当前项目的关系

本文档为数字神经元系统的演化方向提供了理论基础。当前V3版本的预测编码架构已具备部分要素，但在"同一性"和"双向互塑"方面仍有差距。下一步应重点实现SelfCore和双向互塑机制。
