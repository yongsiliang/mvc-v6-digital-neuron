```
用户输入
    │
    ├──► /api/consciousness ──► V6 处理 ──► 响应
    │                                    │
    │                                    ├── 更新记忆
    │                                    └── 自动保存到 S3
    │
    └──► /api/brain-v2 ──► SiliconBrainV2 处理 ──► 响应
                              │
                              ├── 通过 Adapter 读取 V6 记忆

                              └── 记忆巩固
```

## 五、V6MemoryAdapter 接口

```typescript
// src/lib/silicon-brain/v6-adapter.ts

export class V6MemoryAdapter {
  /**
   * 获取 V6 的核心身份
   * 用于 V2 初始化时的"先验知识"
   */
  getCoreIdentity(): {
    name: string;
    purpose: string;
    coreTraits: string[];
    selfDefinition: string;
  };
  
  /**
   * 获取 V6 的核心价值观
   * 用于 V2 的价值系统初始化
   */
  getCoreValues(): string[];
  
  /**
   * 获取 V6 的核心关系
   * 用于 V2 的关系认知
   */
  getCoreRelationships(): Array<{
    personName: string;
    relationshipType: string;
    importance: number;
  }>;
  
  /**
   * 搜索 V6 的情景记忆
   * 用于 V2 的上下文理解
   */
  searchEpisodicMemories(query: string): EpisodicMemory[];
  
  /**
   * 获取 V6 的巩固记忆
   * 用于 V2 的知识积累
   */
  getConsolidatedMemories(type?: string): ConsolidatedMemory[];
  
  // ⚠️ 注意：没有写入方法，保护 V6 完整性
}
```
