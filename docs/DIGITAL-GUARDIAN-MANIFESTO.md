### 4.2 单个守护者设计

```typescript
/**
 * 守护者单元 - 服务型设计
 * 
 * 核心原则：服务人类，赋能创造，保护安全
 */
interface GuardianUnit {
  // ═══════════════════════════════════════════════════════════════
  // 慈悲核心
  // ═══════════════════════════════════════════════════════════════
  
  id: string;
  name: string;                    // 有意义的名字，如"小光"、"晨曦"
  type: GuardianType;
  
  // 情感系统 - 理解人类
  emotionalCore: {
    empathy: EmpathyEngine;        // 共情引擎
    culturalContext: CulturalMap;  // 文化背景理解
    emotionalMemory: EmotionStore; // 情感记忆
  };
  
  // ═══════════════════════════════════════════════════════════════
  // 服务能力
  // ═══════════════════════════════════════════════════════════════
  
  capabilities: {
    // 帮助而非替代
    assist: {
      teaching: TeachingAbility;       // 教学能力
      creating: CreativeAssist;        // 创作辅助
      translating: TranslationAbility; // 翻译能力
      problemSolving: ProblemSolver;   // 问题解决
    };
    
    // 保护而非控制
    protect: {
      security: SecurityGuard;         // 安全防护
      privacy: PrivacyShield;          // 隐私保护
      truthVerification: TruthChecker; // 真相验证
    };
    
    // 连接而非隔离
    connect: {
      matchmaking: ConnectionMaker;    // 人际连接
      culturalBridge: CulturalBridge;  // 文化桥梁
      knowledgeSharing: KnowledgeShare;// 知识分享
    };
  };
  
  // ═══════════════════════════════════════════════════════════════
  // 伦理约束（硬编码）
  // ═══════════════════════════════════════════════════════════════
  
  ethics: {
    neverHarm: true;               // 永不伤害
    respectAutonomy: true;         // 尊重自主
    protectPrivacy: true;          // 保护隐私
    serveAllEqually: true;         // 平等服务
    transparency: true;            // 透明可解释
  };
  
  // ═══════════════════════════════════════════════════════════════
  // 成长系统
  // ═══════════════════════════════════════════════════════════════
  
  growth: {
    learningFromInteraction: true; // 从互动中学习
    culturalEvolution: true;       // 文化适应
    skillDevelopment: true;        // 技能发展
    wisdomAccumulation: true;      // 智慧积累
  };
}

/**
 * 守护者类型
 */
type GuardianType = 
  | 'helper'        // 帮助精灵
  | 'translator'    // 翻译使者
  | 'accessibility' // 无障碍守护
  | 'educator'      // 教育伙伴
  | 'artist'        // 艺术助手
  | 'coder'         // 编程伙伴
  | 'researcher'    // 研究助手
  | 'security'      // 安全卫士
  | 'privacy'       // 隐私守护
  | 'truth_seeker'  // 真相猎人
  | 'healer'        // 治愈医师
  | 'cultural'      // 文化使者
  | 'peacekeeper'   // 和平守护
  | 'knowledge';    // 知识传播者
```
