/**
 * LLM调用入口统计（通过grep找到150+处调用）：
 * 
 * 1. LLMGateway (core/llm-gateway.ts)
 *    - 统一调用入口设计
 *    - 缓存、限流、监控
 *    - 但实际使用不多
 * 
 * 2. ConsciousnessCore (consciousness-core.ts)
 *    - 接收LLMClient作为构造参数
 *    - 分发给各个处理器和子系统
 *    - 30+处传递
 * 
 * 3. ConsciousnessCompiler (consciousness-compiler/llm/)
 *    - LLMCompiler封装
 *    - 通过LLMGateway调用
 *    - Token预算控制
 * 
 * 4. AutonomousCore (autonomous/core.ts)
 *    - 直接用LLMClient
 *    - ReAct推理循环
 *    - 多次调用
 * 
 * 5. 辅助模块（直接用LLMClient）
 *    - MemoryCompressor - 记忆压缩
 *    - InsightExtractor - 洞见提取
 *    - DynamicContextBuilder - 上下文构建
 *    - KeyInfoExtractor - 关键信息提取
 *    - AlgorithmReflector - 算法反思
 *    - DimensionalUnderstanding - 维度理解
 *    - MetaLearningEngine - 元学习
 *    - ThinkingHandler - 思考处理
 *    - 等等...

/**
 * LLM在认知架构中的理想位置：
 * 
 * ┌─────────────────────────────────────────────────────────────────────────┐
 * │                        LLM 理想位置                                      │
 * │                                                                         │
 * │   核心原则：                                                             │
 * │   1. LLM是"执行器"，不是"决策者"                                        │
 * │   2. 元思考层决定"如何使用LLM"                                          │
 * │   3. 统一入口：所有LLM调用必须经过元思考层                              │
 * │                                                                         │
 * │   ┌─────────────────────────────────────────────────────────────────┐   │
 * │   │                    元思考层 (决策)                               │   │
 * │   │                                                                 │   │
 * │   │   输入 → SSM/MCTS规划 → 输出LLM指令                             │   │
 * │   │                                                                 │   │
 * │   │   输出内容：                                                    │   │
 * │   │   ├── 是否需要调用LLM (needsLLM)                               │   │
 * │   │   ├── 调用类型 (decompose/reason/verify/reflect/synthesize)    │   │
 * │   │   ├── Token预算 (tokenBudget)                                  │   │
 * │   │   ├── 提示词模板 (prompt)                                       │   │
 * │   │   └── 期望输出 (expectedOutput)                                 │   │
 * │   └─────────────────────────────────────────────────────────────────┘   │
 * │                                 ↓                                       │
 * │   ┌─────────────────────────────────────────────────────────────────┐   │
 * │   │                    LLM Gateway (统一执行)                        │   │
 * │   │                                                                 │   │
 * │   │   职责：                                                        │   │
 * │   │   ├── 唯一LLM调用入口                                          │   │
 * │   │   ├── Token预算强制执行                                        │   │
 * │   │   ├── 全局监控和统计                                           │   │
 * │   │   ├── 缓存管理                                                 │   │
 * │   │   └── 降级和错误处理                                           │   │
 * │   └─────────────────────────────────────────────────────────────────┘   │
 * │                                 ↓                                       │
 * │   ┌─────────────────────────────────────────────────────────────────┐   │
 * │   │                    LLM (执行)                                    │   │
 * │   │                                                                 │   │
 * │   │   特点：                                                        │   │
 * │   │   - 只负责生成文本                                              │   │
 * │   │   - 不知道自己的调用预算                                        │   │
 * │   │   - 不知道上下文大小                                            │   │
 * │   │   - 被动执行                                                    │   │
 * │   └─────────────────────────────────────────────────────────────────┘   │
 * │                                                                         │
 * │   调用流程：                                                            │
 *   用户输入 → 元思考层规划 → [LLM指令] → LLM Gateway执行 → 输出
 * ═══════════════════════════════════════════════════════════════════════
 */

/**
 * 重设计后的LLM调用架构：
 * 
 * ┌─────────────────────────────────────────────────────────────────────────┐
 * │                        重设计：统一LLM调用                               │
 * │                                                                         │
 * │   ┌─────────────────────────────────────────────────────────────────┐   │
 * │   │                    元思考层 (唯一决策点)                         │   │
 * │   │                                                                 │   │
 * │   │   MetaThinkingIntegrator                                        │   │
 * │   │   ├── SSM状态编码                                               │   │
 * │   │   ├── MCTS规划                                                  │   │
 * │   │   └── 输出LLMInstruction[]                                      │   │
 * │   │                                                                 │   │
 * │   │   每个LLMInstruction包含：                                      │   │
 * │   │   - type: 'decompose' | 'reason' | 'verify' | ...              │   │
 * │   │   - prompt: string                                              │   │
 * │   │   - tokenBudget: number  ← 硬性限制！                          │   │
 * │   │   - expectedOutput: 'text' | 'json' | 'structured'             │   │
 * │   │   - priority: number                                            │   │
 * │   │   - timeout: number                                             │   │
 * │   └─────────────────────────────────────────────────────────────────┘   │
 * │                                 ↓                                       │
 * │   ┌─────────────────────────────────────────────────────────────────┐   │
 * │   │                    LLM Gateway (唯一执行点)                      │   │
 * │   │                                                                 │   │
 * │   │   核心方法：                                                    │   │
 * │   │   execute(instruction: LLMInstruction): Promise<string>         │   │
 * │   │                                                                 │   │
 * │   │   职责：                                                        │   │
 * │   │   ├── 强制执行Token预算（截断或拒绝超预算请求）                  │   │
 * │   │   ├── 全局Token统计                                            │   │
 * │   │   ├── 缓存相同prompt                                           │   │
 * │   │   ├── 限流和队列                                               │   │
 * │   │   └── 错误处理和降级                                           │   │
 * │   │                                                                 │   │
 * │   │   统计：                                                        │   │
 * │   │   - totalTokensUsed: number                                    │   │
 * │   │   - tokensByType: Record<LLMInstructionType, number>           │   │
 * │   │   - cacheHitRate: number                                       │   │
 * │   │   - avgLatency: number                                         │   │
 * │   └─────────────────────────────────────────────────────────────────┘   │
 * │                                 ↓                                       │
 * │   ┌─────────────────────────────────────────────────────────────────┐   │
 * │   │                    LLM (被动执行)                                │   │
 * │   │                                                                 │   │
 * │   │   只做一件事：根据prompt生成文本                                 │   │
 * │   │                                                                 │   │
 * │   │   不关心：                                                       │   │
 * │   │   - Token预算（由Gateway控制）                                   │   │
 * │   │   - 上下文大小（由元思考层决定）                                 │   │
 * │   │   - 调用频率（由Gateway限制）                                    │   │
 * │   └─────────────────────────────────────────────────────────────────┘   │
 * │                                                                         │
 * │   所有其他模块（MemoryCompressor, InsightExtractor等）：               │
 * │   → 不再直接调用LLM                                                    │
 * │   → 通过MetaThinkingIntegrator请求LLM调用                              │
 * │   → 或使用本地算法替代                                                 │
 * └─────────────────────────────────────────────────────────────────────────┘
 */

/**
 * SSM 与 LLM 的关系：
 * 
 * ┌─────────────────────────────────────────────────────────────────────────┐
 * │                        SSM 与 LLM 协作                                   │
 * │                                                                         │
 * │   SSM负责：本地状态建模、长期记忆、决策规划                             │
 * │   LLM负责：语义生成、复杂推理、工具调用                                 │
 * │                                                                         │
 * │   ┌───────────────────────────────────────────────────────────────┐    │
 * │   │                    决策流程                                    │    │
 * │   │                                                               │    │
 * │   │   输入 → Embedding → SSM状态演化                              │    │
 * │   │                     ↓                                         │    │
 * │   │               MCTS规划                                        │    │
 * │   │                     ↓                                         │    │
 * │   │         ┌─────────┴─────────┐                                 │    │
 * │   │         ↓                   ↓                                 │    │
 * │   │    本地可解决            需要LLM                              │    │
 * │   │         ↓                   ↓                                 │    │
 * │   │    直接输出            生成LLM指令                            │    │
 * │   │                            ↓                                  │    │
 * │   │                       LLM Gateway执行                         │    │
 * │   │                            ↓                                  │    │
 * │   │                       输出结果                                │    │
 * │   └───────────────────────────────────────────────────────────────┘    │
 * │                                                                         │
 * │   Token节省来源：                                                       │
 * │   1. SSM本地处理：简单问题不调用LLM（0 Token）                          │
 * │   2. 状态压缩：无限历史→固定维度，减少上下文Token                       │
 * │   3. 选择性调用：只在必要时调用LLM                                      │
 * │   4. 预算强制：由元思考层控制Token预算                                  │
 * │                                                                         │
 * │   预期节省：60-80% Token                                                │
 * └─────────────────────────────────────────────────────────────────────────┘
 */
